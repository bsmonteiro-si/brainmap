```tsx
import { useState, useMemo } from "react";
import { useGraphStore } from "../../stores/graphStore";
import { useEditorStore } from "../../stores/editorStore";
import { useUIStore } from "../../stores/uiStore";

interface TreeNode {
  name: string;
  fullPath: string;    // folder: "concepts", "concepts/sub"; file: rel path
  title: string;
  isFolder: boolean;
  children: TreeNode[];
}

function buildTree(nodes: Map<string, { path: string; title: string }>): TreeNode[] {
  const root: Record<string, TreeNode> = {};

  for (const [path, node] of nodes) {
    const parts = path.replace(/\.md$/, "").split("/");
    let currentLevel = root;
    let cumulativePath = "";

    for (let i = 0; i < parts.length - 1; i++) {
      cumulativePath = cumulativePath ? `${cumulativePath}/${parts[i]}` : parts[i];
      if (!currentLevel[parts[i]]) {
        currentLevel[parts[i]] = {
          name: parts[i],
          fullPath: cumulativePath,
          title: parts[i],
          isFolder: true,
          children: [],
        };
      }
      // Navigate into children map (we'll sort children at the end)
      const childMap: Record<string, TreeNode> = {};
      for (const c of currentLevel[parts[i]].children) {
        childMap[c.name] = c;
      }
      currentLevel = childMap;
    }

    const leaf = parts[parts.length - 1];
    currentLevel[leaf] = {
      name: leaf,
      fullPath: path,
      title: node.title,
      isFolder: false,
      children: [],
    };
  }

  // Flatten root record to sorted array (folders first)
  function toSorted(record: Record<string, TreeNode>): TreeNode[] {
    return Object.values(record).sort((a, b) => {
      if (a.isFolder !== b.isFolder) return a.isFolder ? -1 : 1;
      return a.name.localeCompare(b.name);
    });
  }

  return toSorted(root);
}
```

**Simpler implementation note**: The nested-map approach above has a bug (building a new `childMap` each iteration loses previous siblings). Use a flat index instead:

```tsx
function buildTree(nodes: Map<string, { path: string; title: string }>): TreeNode[] {
  // Map from full folder path → TreeNode (for folders)
  const folderMap = new Map<string, TreeNode>();
  const roots: TreeNode[] = [];

  function getOrCreateFolder(parts: string[], depth: number): TreeNode {
    const fullPath = parts.slice(0, depth + 1).join("/");
    if (!folderMap.has(fullPath)) {
      const node: TreeNode = {
        name: parts[depth],
        fullPath,
        title: parts[depth],
        isFolder: true,
        children: [],
      };
      folderMap.set(fullPath, node);
      if (depth === 0) {
        roots.push(node);
      } else {
        const parentPath = parts.slice(0, depth).join("/");
        folderMap.get(parentPath)!.children.push(node);
      }
    }
    return folderMap.get(fullPath)!;
  }

  for (const [path, node] of nodes) {
    const parts = path.replace(/\.md$/, "").split("/");

    if (parts.length === 1) {
      // Root-level file (no folder)
      roots.push({ name: parts[0], fullPath: path, title: node.title, isFolder: false, children: [] });
    } else {
      // Build folder chain
      for (let i = 0; i < parts.length - 1; i++) {
        getOrCreateFolder(parts, i);
      }
      const parentPath = parts.slice(0, parts.length - 1).join("/");
      folderMap.get(parentPath)!.children.push({
        name: parts[parts.length - 1],
        fullPath: path,
        title: node.title,
        isFolder: false,
        children: [],
      });
    }
  }

  // Sort: folders first, then alphabetically
  function sortChildren(nodes: TreeNode[]): TreeNode[] {
    return nodes
      .sort((a, b) => {
        if (a.isFolder !== b.isFolder) return a.isFolder ? -1 : 1;
        return a.name.localeCompare(b.name);
      })
      .map(n => ({ ...n, children: sortChildren(n.children) }));
  }

  return sortChildren(roots);
}
```

```tsx
function FileTreeNode({ node, depth }: { node: TreeNode; depth: number }) {
  const selectedNodePath = useGraphStore(s => s.selectedNodePath);
  const treeExpandedFolders = useUIStore(s => s.treeExpandedFolders);
  const toggleFolder = useUIStore(s => s.toggleFolder);

  if (node.isFolder) {
    const isExpanded = treeExpandedFolders.has(node.fullPath);
    return (
      <div>
        <div
          className="tree-item tree-folder"
          style={{ paddingLeft: `${8 + depth * 16}px` }}
          onClick={() => toggleFolder(node.fullPath)}
        >
          <span className="tree-icon">{isExpanded ? "▾" : "▸"}</span>
          {node.name}
        </div>
        {isExpanded && node.children.map(c => (
          <FileTreeNode key={c.fullPath} node={c} depth={depth + 1} />
        ))}
      </div>
    );
  }

  const isActive = selectedNodePath === node.fullPath;
  return (
    <div
      className={`tree-item tree-file ${isActive ? "active" : ""}`}
      style={{ paddingLeft: `${8 + depth * 16}px` }}
      onClick={() => {
        useGraphStore.getState().selectNode(node.fullPath);
        useEditorStore.getState().openNote(node.fullPath);
      }}
    >
      <span className="tree-icon">📄</span>
      {node.title}
    </div>
  );
}

export function FileTreePanel() {
  const nodes = useGraphStore(s => s.nodes);
  const [filter, setFilter] = useState("");

  const tree = useMemo(() => buildTree(nodes), [nodes]);

  // Filter: hide non-matching leaves; show parent folders if any child matches
  const filtered = filter.trim()
    ? filterTree(tree, filter.toLowerCase())
    : tree;

  return (
    <div className="file-tree-panel">
      <div className="file-tree-filter">
        <input
          placeholder="Filter..."
          value={filter}
          onChange={e => setFilter(e.target.value)}
        />
      </div>
      <div className="file-tree-content">
        {filtered.map(n => <FileTreeNode key={n.fullPath} node={n} depth={0} />)}
      </div>
    </div>
  );
}

function filterTree(nodes: TreeNode[], q: string): TreeNode[] {
  return nodes.flatMap(n => {
    if (!n.isFolder) {
      return n.title.toLowerCase().includes(q) ? [n] : [];
    }
    const filteredChildren = filterTree(n.children, q);
    return filteredChildren.length > 0 ? [{ ...n, children: filteredChildren }] : [];
  });
}
```

```tsx
// crates/app/src/components/Editor/RelatedNotesFooter.tsx
import { useMemo, useState } from "react";
import { useEditorStore } from "../../stores/editorStore";
import { useGraphStore } from "../../stores/graphStore";

export function RelatedNotesFooter() {
  const activeNote = useEditorStore((s) => s.activeNote);
  const edges = useGraphStore((s) => s.edges);
  const nodes = useGraphStore((s) => s.nodes);
  const [expanded, setExpanded] = useState(false);
  const [showAll, setShowAll] = useState(false);

  const related = useMemo(() => {
    if (!activeNote) return [];
    const outgoing = (activeNote.links ?? []).map((l) => ({
      dir: "out" as const,
      rel: l.rel,
      path: l.target,
      title: nodes.get(l.target)?.title ?? l.target,
    }));
    const incoming = edges
      .filter((e) => e.target === activeNote.path)
      .map((e) => ({
        dir: "in" as const,
        rel: e.rel,
        path: e.source,
        title: nodes.get(e.source)?.title ?? e.source,
      }));
    return [...outgoing, ...incoming];
  }, [activeNote, edges, nodes]);

  if (!activeNote || related.length === 0) return null;

  const visible = showAll ? related : related.slice(0, 10);

  return (
    <div className="related-notes-footer">
      <button
        className="related-notes-toggle"
        onClick={() => setExpanded((e) => !e)}
      >
        <span>{expanded ? "▾" : "▸"}</span>
        Related Notes ({related.length})
      </button>
      {expanded && (
        <div className="related-notes-list">
          {visible.map((item, i) => (
            <div
              key={i}
              className="related-note-item"
              onClick={() => {
                useGraphStore.getState().selectNode(item.path);
                useEditorStore.getState().openNote(item.path);
              }}
            >
              <span className="related-note-dir">{item.dir === "out" ? "→" : "←"}</span>
              <span className="related-note-rel">{item.rel}</span>
              <span className="related-note-title">{item.title}</span>
            </div>
          ))}
          {!showAll && related.length > 10 && (
            <button
              className="related-notes-more"
              onClick={() => setShowAll(true)}
            >
              Show {related.length - 10} more…
            </button>
          )}
        </div>
      )}
    </div>
  );
}
```

`activeNote.links` is typed from `editorStore`. Check `NoteDetail` type in `api/types.ts` — links are `{ target: string; rel: string; kind: string }[]`. Use `l.target` for path and `l.rel` for the relationship.

### 6b. `GraphView.tsx` — add dagre layout support

At top:
```ts
import dagre from "cytoscape-dagre";
cytoscape.use(dagre);
```

The layout-run effect needs to know which layout to use. Pass `graphLayout` from the store via a ref. When `graphLayout === "hierarchical"`, run dagre on filtered elements:

```ts
const DIRECTIONAL_RELS = new Set(["precedes", "leads-to", "causes", "extends", "depends-on", "evolved-from", "part-of", "contains"]);

function runLayout(cy: Core, layout: "force" | "hierarchical") {
  if (layout === "hierarchical") {
    const layoutElements = cy.elements().filter((el) => {
      if (el.isNode()) return true;
      return DIRECTIONAL_RELS.has(el.data("label") as string);
    });
    (layoutElements as cytoscape.Collection).layout({
      name: "dagre",
      rankDir: "LR",
      nodeSep: 60,
      rankSep: 120,
      animate: false,
      fit: true,
      padding: 40,
    } as cytoscape.LayoutOptions).run();
  } else {
    cy.layout({
      name: "fcose",
      animate: false,
      quality: "proof",
      idealEdgeLength: 120,
      nodeRepulsion: 8000,
      gravity: 0.25,
      fit: true,
      padding: 40,
      nodeDimensionsIncludeLabels: true,
    } as cytoscape.LayoutOptions).run();
  }
}
```

Replace the `cy.layout(...)` call in the data-sync effect with `runLayout(cy, graphLayoutRef.current)`.

Also add a separate `useEffect` that watches `graphLayout` store value and re-runs the layout without replacing all elements:
```ts
useEffect(() => {
  const cy = cyRef.current;
  if (!cy || cy.nodes().length === 0) return;
  runLayout(cy, graphLayout);
}, [graphLayout]);
```
