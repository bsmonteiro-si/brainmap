     - Note focus: focal node + direct neighbors + edges between them; always adds `focusPath` itself to `neighborPaths` to handle isolated nodes
     - Folder focus: all notes with path starting with `focusPath + "/"` + intra-folder edges only (cross-folder excluded by design)
     ```typescript
     export function filterGraphByFocus(
       nodes: Map<string, NodeDto>,
       edges: EdgeDto[],
       focusPath: string,
       focusKind: "note" | "folder"
     ): FocusFilterResult {
       if (focusKind === "note") {
         const connectedEdges = edges.filter(e => e.source === focusPath || e.target === focusPath);
         const neighborPaths = new Set(connectedEdges.flatMap(e => [e.source, e.target]));
         neighborPaths.add(focusPath); // always include focal node even if no edges
         return { filteredNodes: [...nodes.values()].filter(n => neighborPaths.has(n.path)), filteredEdges: connectedEdges, focalPath: focusPath };
       } else {
         const prefix = focusPath + "/";
         const folderNodes = [...nodes.values()].filter(n => n.path.startsWith(prefix));
         const folderPaths = new Set(folderNodes.map(n => n.path));
         return { filteredNodes: folderNodes, filteredEdges: edges.filter(e => folderPaths.has(e.source) && folderPaths.has(e.target)), focalPath: null };
       }
     }
     ```
